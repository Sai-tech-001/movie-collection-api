Browse the movies using the buttons below.

style.css body { font-family: Arial, sans-serif; padding: 20px; } .buttons { margin-bottom: 20px; } button { margin-right: 10px; padding: 10px 15px; cursor: pointer; } #output .movie, #output .genre { border: 1px solid #ccc; padding: 15px; margin-bottom: 10px; border-radius: 5px; } .classic { background-color: gold; color: black; padding: 3px 6px; border-radius: 3px; font-size: 12px; } script.js const output = document.getElementById("output"); document.getElementById("allBtn").addEventListener("click", () => { fetch("/movies") .then(res => res.json()) .then(data => displayMovies(data)) .catch(() => showError()); }); document.getElementById("classicBtn").addEventListener("click", () => { fetch("/movies/classics") .then(res => res.json()) .then(data => displayMovies(data)) .catch(() => showError()); }); document.getElementById("genreBtn").addEventListener("click", () => { fetch("/movies/genres") .then(res => res.json()) .then(data => displayGenres(data.genres)) .catch(() => showError()); }); function displayMovies(movies) { output.innerHTML = ""; movies.forEach(movie => { const div = document.createElement("div"); div.className = "movie"; div.innerHTML = `

${movie.title} ${movie.isClassic ? 'Classic' : ''}

Genre: ${movie.genre}

Release Year: ${movie.releaseYear}

Director: ${movie.director}

`; output.appendChild(div); }); } function displayGenres(genres) { output.innerHTML = ""; genres.forEach(g => { const div = document.createElement("div"); div.className = "genre"; div.innerHTML = `

${g.name}

Movies: ${g.movieCount}

`; output.appendChild(div); }); } function showError() { output.innerHTML = "

Error loading data.

"; }
